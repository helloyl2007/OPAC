async def text_to_speech(text: str, output_path: str, voice: str = "jielidou") -> bool:
    """
    将文本转换为语音
    
    参数:
        text: 要转换的文本
        output_path: 输出语音文件路径
        voice: 音色名称，直接使用阿里云音色名称，如 aixia、xiaoxiao、sicheng 等
        
    返回:
        成功状态
    """
    try:
        logger.info(f"开始将文本转换为语音: {text[:30]}...，使用音色: {voice}")
        
        # 确保输出目录存在
        os.makedirs(os.path.dirname(output_path), exist_ok=True)
        
        # 检查是否配置了阿里云TTS
        if ALI_TTS_CONFIG['appKey'] and ALI_TTS_CONFIG['token']:
            # 使用阿里云TTS服务
            success = await ali_tts(text, output_path, voice.lower())
            if success:
                logger.info(f"阿里云TTS语音生成成功: {output_path}")
                return True
            else:
                logger.error("阿里云TTS语音生成失败，将尝试备选方案")
        else:
            logger.warning("未配置阿里云TTS参数，将尝试备选方案")

        # 备选方案1: 尝试使用边缘TTS (edge-tts)
        try:
            from edge_tts import Communicate
            
            # 对于 edge-tts，我们需要将音色映射回微软格式
            ms_voice_map = {
                'xiaoxiao': 'zh-CN-XiaoxiaoNeural',
                'aixia': 'zh-CN-YunyangNeural', 
                'sicheng': 'zh-CN-YunjianNeural',
                'xiaowei': 'zh-CN-XiaoyiNeural',
                'ninger': 'zh-CN-YunxiNeural'
            }
            
            edge_voice = ms_voice_map.get(voice.lower(), 'zh-CN-YunyangNeural')
            communicate = Communicate(text, edge_voice)
            await communicate.save(output_path)
            
            logger.info(f"edge-tts 语音生成成功: {output_path}")
            return True
            
        except ImportError:
            logger.warning("edge-tts库未安装，将生成默认语音文件")
            
            # 备选方案2: 生成一个简单的静音音频文件
            duration = max(5, len(text) / 5)  # 简单估算时长：每5个字符1秒
            try:
                process = await asyncio.create_subprocess_exec(
                    "ffmpeg", "-y", "-f", "lavfi", "-i", f"anullsrc=r=44100:cl=mono", 
                    "-t", str(duration), output_path,
                    stdout=asyncio.subprocess.PIPE,
                    stderr=asyncio.subprocess.PIPE
                )
                
                stdout, stderr = await process.communicate()
                
                if process.returncode != 0:
                    logger.error(f"FFmpeg错误: {stderr.decode()}")
                    return False
                    
                logger.info(f"创建静音音频成功: {output_path}")
                return True
                
            except Exception as e:
                logger.error(f"创建音频文件失败: {str(e)}")
                return False
        
    except Exception as e:
        logger.error(f"文本转语音失败: {str(e)}")
        return False

async def ali_tts(text: str, output_path: str, voice: str) -> bool:
    """
    使用阿里云TTS服务将文本转换为语音
    
    参数:
        text: 要转换的文本
        output_path: 输出语音文件路径
        voice: 阿里云音色名称，如 xiaoxiao, aixia, sicheng 等
        
    返回:
        成功状态
    """
    try:
        # 直接使用提供的音色名称，如果未提供则默认使用 aixia
        ali_voice = voice if voice else 'aixia'
        
        # 检查音色是否在支持列表中
        if ali_voice not in VOICE_OPTIONS:
            logger.warning(f"未知的音色名称: {ali_voice}，将使用默认音色 aixia")
            ali_voice = 'aixia'
            
        logger.info(f"使用阿里云音色: {ali_voice} ({VOICE_OPTIONS.get(ali_voice, '未知音色')})")
        
        # URL编码处理文本
        text_urlencode = urllib.parse.quote_plus(text)
        text_urlencode = text_urlencode.replace("+", "%20")
        text_urlencode = text_urlencode.replace("*", "%2A")
        text_urlencode = text_urlencode.replace("%7E", "~")
        
        # 获取配置
        app_key = ALI_TTS_CONFIG['appKey']
        token = ALI_TTS_CONFIG['token']
        host = ALI_TTS_CONFIG['host']
        format = ALI_TTS_CONFIG['format']
        sample_rate = ALI_TTS_CONFIG['sample_rate']
        
        # 创建连接
        conn = http.client.HTTPSConnection(host)
        
        # 设置请求头
        http_headers = {
            'Content-Type': 'application/json'
        }
        
        # 设置请求体
        body = {
            'appkey': app_key, 
            'token': token, 
            'text': text, 
            'format': format, 
            'sample_rate': sample_rate,
            'voice': ali_voice
        }
        body_json = json.dumps(body)
        
        # 发送请求
        url = f'/stream/v1/tts'
        conn.request(method='POST', url=url, body=body_json, headers=http_headers)
        
        # 处理响应
        response = conn.getresponse()
        logger.info(f"阿里云TTS响应状态: {response.status} {response.reason}")
        
        content_type = response.getheader('Content-Type')
        response_body = response.read()
        
        # 检查响应是否是音频
        if 'audio/mpeg' == content_type or 'audio/wav' == content_type:
            with open(output_path, mode='wb') as f:
                f.write(response_body)
            logger.info('阿里云TTS请求成功!')
            conn.close()
            return True
        else:
            error_msg = response_body.decode('utf-8')
            logger.error(f'阿里云TTS请求失败: {error_msg}')
            conn.close()
            return False
            
    except Exception as e:
        logger.error(f"阿里云TTS服务调用失败: {str(e)}")
        return False
